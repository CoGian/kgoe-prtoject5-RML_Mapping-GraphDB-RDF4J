PREFIX ex: <http://www.semanticweb.org/kgiantsios/ontologies/2022/5/imdb#>
SELECT ?title ?total
WHERE {
    ?movie a ex:Movie;
    		ex:primaryTitle ?title.
	?rating ex:ratingFor ?movie.
	?rating ex:total ?total
}
ORDER BY DESC(?total)
LIMIT 1
